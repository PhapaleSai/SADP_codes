//Write a Java Program to implement command pattern to test Remote Control 

// Command interface
interface Command { void execute(); }

// Receiver
class Light {
    void on() { System.out.println("Light is ON"); }
    void off() { System.out.println("Light is OFF"); }
}

// Invoker
class Remote {
    private Command cmd;
    void setCommand(Command c) { cmd = c; }
    void press() { cmd.execute(); }
}

// Concrete Commands
class LightOn implements Command {
    private Light l;
    LightOn(Light l) { this.l = l; }
    public void execute() { l.on(); }
}
class LightOff implements Command {
    private Light l;
    LightOff(Light l) { this.l = l; }
    public void execute() { l.off(); }
}

// Client
public class CommandDemo {
    public static void main(String[] args) {
        Light light = new Light();
        Remote r = new Remote();

        r.setCommand(new LightOn(light));  r.press();
        r.setCommand(new LightOff(light)); r.press();
    }
}

// Output:
// Light is ON
// Light is OFF
